3. Поиск элемента по селектору querySelector( )

        
// возвращает первый найденный элемент
document.querySelector('#auth')
document.querySelector('.auth')

// возвращает коллекцию элементов
document.querySelectorAll('.auth')
document.querySelectorAll('input, button')

5. Получение значений атрибутов

        
form.getAttribute('action'); // значение атрибута action
form.hasAttribute('method'); // наличие атрибута method true/false
form.setAttribute('method', 'POST'); // наличие атрибута method true/false
form.removeAttribute('method'); // удаление атрибута по имени

6. Дата-атрибуты

Пример:

        
<div data-form-value="123"></div>

Методы:

        
form.dataset.formValue; // получение значения атрибута data-form-value
form.dataset.hasOwnProperty('formValue'); // проверка наличия атрибута data-form-value, true/false
form.dataset.formValue = '321'; // установить значение атрибута
form.hasAttribute('formValue'); // наличие дата-атрибута method true/false

7. Классы classList

        
            form.classList.add('login'); // добавить класс
            form.classList.item(1); // получение класса по индексу из списка классов
            form.classList.contains('login'); // проверка наличия класса, true/false
            form.classList.remove('login'); // удаление класса

8. Создание DOM-элемента

        
// создаем элемент
let elem = document.createElement('span');

// наделяем свойствами
elem.classList.add('error');
elem.textContent = 'Введен неверный логин';

// добавляем элемент в структуру DOM как вложенный дочерний элемент body
document.body.appendChild(elem);

9. Копирование DOM-элемента

        
let clone = elem.cloneNode(true); // true - копирует все дочерние элементы, false - не копирует

10. События

        
form.addEventListener('submit', submitHandler); // добавляем на submit слушателя submitHandler*

* 1 событие = 1 обработчик.

Список обработки событий:

        
button.addEventListener('submit', function (event) {
    console.log(this, event);
});
button.addEventListener('submit', clickHandler);
button.addEventListener('submit', twoClick);

Отмена события по умолчанию:

        
event.preventDefault();
