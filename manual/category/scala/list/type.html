Типы

        
            // объявление переменных с указанием типа
val b0: Byte = 0
val b1: Byte = 0x7F // 127
val s0: Short = 127
val s1: Short = 255

print(s0.toByte) // 127
print(s1.toByte) // -1, т.к. 255 за границами Byte

// объявление переменных без указания типа
val l = 1000L // long
val i = 0x7F // int
val d = 0.0 // double
val f = 0.0F // float
val ch = 'A' // char
val str = "Some" // string
